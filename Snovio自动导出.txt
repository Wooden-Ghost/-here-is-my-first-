// Snov.io 自动化导出工具 - 修正版
// 复制以下所有代码，在Chrome/Firefox控制台中粘贴并运行

(function() {
    'use strict';
    
    console.log('🎯 Snov.io 自动化导出工具 - 修正版');
    
    // 移除旧面板
    const oldPanel = document.getElementById('snov-auto-export-panel');
    if (oldPanel) oldPanel.remove();
    
    // 创建面板
    const panel = document.createElement('div');
    panel.id = 'snov-auto-export-panel';
    panel.style.cssText = `
        position: fixed;
        top: 20px;
        right: 20px;
        width: 300px;
        background: white;
        border-radius: 8px;
        box-shadow: 0 4px 20px rgba(0,0,0,0.15);
        z-index: 999999;
        font-family: Arial, sans-serif;
        border: 1px solid #ddd;
        overflow: hidden;
    `;
    
    // 标题栏
    const titleBar = document.createElement('div');
    titleBar.style.cssText = `
        background: #4a6ee0;
        color: white;
        padding: 10px 12px;
        display: flex;
        justify-content: space-between;
        align-items: center;
        cursor: move;
    `;
    titleBar.innerHTML = `
        <div style="font-weight: bold; font-size: 14px;">Snov.io 自动导出</div>
        <div id="status-indicator" style="background: #4CAF50; padding: 2px 8px; border-radius: 10px; font-size: 11px;">就绪</div>
        <div style="display: flex; gap: 5px;">
            <button id="close-btn" style="background: rgba(255,255,255,0.2); border: none; color: white; width: 20px; height: 20px; border-radius: 50%; cursor: pointer; font-size: 14px;">×</button>
        </div>
    `;
    
    // 内容区
    const content = document.createElement('div');
    content.style.cssText = `
        padding: 12px;
    `;
    content.innerHTML = `
        <button id="run-export" style="width: 100%; padding: 10px; background: #4CAF50; color: white; border: none; border-radius: 5px; cursor: pointer; font-weight: bold; font-size: 13px; margin-bottom: 10px;">
            开始自动导出
        </button>
        <button id="stop-export" style="width: 100%; padding: 10px; background: #f44336; color: white; border: none; border-radius: 5px; cursor: pointer; font-weight: bold; font-size: 13px; display: none;">
            停止
        </button>
        
        <div style="margin-top: 10px; border-top: 1px solid #eee; padding-top: 10px;">
            <div style="font-size: 12px; color: #666; margin-bottom: 5px;">操作日志</div>
            <div id="export-log" style="height: 150px; overflow-y: auto; background: #f8f9fa; border-radius: 5px; padding: 8px; font-family: monospace; font-size: 11px; border: 1px solid #eee;">
                <div>面板已加载</div>
            </div>
        </div>
    `;
    
    panel.appendChild(titleBar);
    panel.appendChild(content);
    document.body.appendChild(panel);
    
    // 拖拽功能
    let isDragging = false;
    let startX, startY, startLeft, startTop;
    
    titleBar.addEventListener('mousedown', function(e) {
        if (e.target.tagName === 'BUTTON') return;
        
        isDragging = true;
        startX = e.clientX;
        startY = e.clientY;
        startLeft = panel.offsetLeft;
        startTop = panel.offsetTop;
        
        document.addEventListener('mousemove', onDrag);
        document.addEventListener('mouseup', stopDrag);
    });
    
    function onDrag(e) {
        if (!isDragging) return;
        
        const dx = e.clientX - startX;
        const dy = e.clientY - startY;
        
        let newLeft = startLeft + dx;
        let newTop = startTop + dy;
        
        const maxX = window.innerWidth - panel.offsetWidth;
        const maxY = window.innerHeight - panel.offsetHeight;
        
        newLeft = Math.max(0, Math.min(newLeft, maxX));
        newTop = Math.max(0, Math.min(newTop, maxY));
        
        panel.style.left = `${newLeft}px`;
        panel.style.top = `${newTop}px`;
        panel.style.right = 'auto';
    }
    
    function stopDrag() {
        isDragging = false;
        document.removeEventListener('mousemove', onDrag);
        document.removeEventListener('mouseup', stopDrag);
    }
    
    // 关闭功能
    document.getElementById('close-btn').addEventListener('click', function() {
        panel.style.display = 'none';
    });
    
    // 日志函数
    function log(message, type = 'info') {
        const time = new Date().toLocaleTimeString();
        const logElement = document.getElementById('export-log');
        
        const logItem = document.createElement('div');
        logItem.style.cssText = `
            padding: 3px 0;
            border-bottom: 1px solid #eee;
            color: ${type === 'error' ? '#f44336' : type === 'success' ? '#4CAF50' : '#333'};
            font-weight: ${type === 'error' ? 'bold' : 'normal'};
        `;
        logItem.textContent = `[${time}] ${message}`;
        
        logElement.appendChild(logItem);
        logElement.scrollTop = logElement.scrollHeight;
        
        console.log(`[自动导出] ${message}`);
    }
    
    // 更新状态
    function updateStatus(status, isRunning = false) {
        const statusEl = document.getElementById('status-indicator');
        const runBtn = document.getElementById('run-export');
        const stopBtn = document.getElementById('stop-export');
        
        statusEl.textContent = status;
        statusEl.style.background = isRunning ? '#ff9800' : '#4CAF50';
        
        runBtn.style.display = isRunning ? 'none' : 'block';
        stopBtn.style.display = isRunning ? 'block' : 'none';
    }
    
    // 等待函数
    function wait(ms) {
        return new Promise(resolve => setTimeout(resolve, ms));
    }
    
    // 等待元素出现
    function waitForElement(selector, timeout = 8000) {
        return new Promise((resolve, reject) => {
            const startTime = Date.now();
            
            const check = () => {
                const element = document.querySelector(selector);
                if (element) {
                    resolve(element);
                    return;
                }
                
                if (Date.now() - startTime > timeout) {
                    reject(new Error(`超时未找到: ${selector}`));
                    return;
                }
                
                setTimeout(check, 500);
            };
            
            check();
        });
    }
    
    // 通过文本查找元素
    function findElementByText(text) {
        const elements = document.querySelectorAll('*');
        for (const el of elements) {
            if (el.textContent && el.textContent.includes(text)) {
                return el;
            }
        }
        return null;
    }
    
    // 安全点击
    function safeClick(element) {
        if (!element) return false;
        
        try {
            element.click();
            return true;
        } catch (e) {
            try {
                element.dispatchEvent(new MouseEvent('click', { bubbles: true }));
                return true;
            } catch (e2) {
                return false;
            }
        }
    }
    
    // 步骤1: 点击导出菜单
    async function clickExportMenu() {
        log('步骤1: 点击导出菜单');
        
        // 查找导出菜单按钮（三个点）
        const selectors = [
            'button.prospects-context-menu__btn',
            'button[data-test="snov-btn"][plerdy-tracking-id="67360642701"]',
            'button.snv-btn--with-icon-only'
        ];
        
        for (const selector of selectors) {
            try {
                const element = document.querySelector(selector);
                if (element) {
                    log(`找到导出菜单: ${selector}`);
                    if (safeClick(element)) {
                        await wait(1500); // 等待菜单展开
                        return true;
                    }
                }
            } catch (e) {
                log(`选择器 ${selector} 失败`, 'error');
            }
        }
        
        throw new Error('未找到导出菜单按钮');
    }
    
    // 步骤2: 点击导出潜在客户（打开导出对话框）
    async function clickExportLeads() {
        log('步骤2: 点击导出潜在客户');
        
        // 方法1: 使用 tracking-id 查找
        let element = document.querySelector('[plerdy-tracking-id="53883892001"]');
        
        // 方法2: 通过文本查找
        if (!element) {
            element = findElementByText('导出潜在客户');
            if (element) {
                element = element.closest('.app-dropdown__drop-item') || element;
            }
        }
        
        // 方法3: 查找所有菜单项
        if (!element) {
            const menuItems = document.querySelectorAll('.app-dropdown__drop-item');
            for (const item of menuItems) {
                if (item.textContent && item.textContent.includes('导出潜在客户')) {
                    element = item;
                    break;
                }
            }
        }
        
        if (!element) {
            throw new Error('未找到导出潜在客户选项');
        }
        
        if (safeClick(element)) {
            log('成功点击导出潜在客户，等待对话框打开...', 'success');
            await wait(2000); // 等待导出对话框完全打开
            return true;
        }
        
        throw new Error('点击导出潜在客户失败');
    }
    
    // 步骤3: 在导出对话框中选择所有邮箱状态
    async function selectEmailStatusesInExportDialog() {
        log('步骤3: 选择邮箱状态 (导出对话框)');
        
        try {
            // 在导出对话框中查找邮箱状态区域 - 使用 .export__type-filters 容器
            const exportFilters = document.querySelector('.export__type-filters');
            if (!exportFilters) {
                throw new Error('未找到导出筛选器容器');
            }
            
            // 在导出筛选器容器中查找邮箱区域
            const emailSection = exportFilters.querySelector('.t-filters__section--email');
            if (!emailSection) {
                throw new Error('未找到导出对话框中的邮箱状态区域');
            }
            
            // 获取所有邮箱状态项
            const emailItems = emailSection.querySelectorAll('.t-filters__item');
            log(`找到 ${emailItems.length} 个邮箱状态选项`);
            
            if (emailItems.length === 0) {
                throw new Error('未找到邮箱状态选项');
            }
            
            // 点击所有未激活的邮箱状态项
            let selectedCount = 0;
            for (const item of emailItems) {
                if (!item.classList.contains('t-filters__item--active')) {
                    if (safeClick(item)) {
                        selectedCount++;
                        await wait(300); // 每个选项间短延迟
                    }
                } else {
                    selectedCount++;
                }
            }
            
            log(`已选择 ${selectedCount}/${emailItems.length} 个邮箱状态`, 'success');
            return selectedCount > 0;
            
        } catch (e) {
            log(`选择邮箱状态失败: ${e.message}`, 'error');
            return false;
        }
    }
    
    // 步骤4: 在导出对话框中选择所有领英资料
    async function selectLinkedInProfilesInExportDialog() {
        log('步骤4: 选择领英资料 (导出对话框)');
        
        try {
            // 在导出对话框中查找领英资料区域 - 使用 .export__type-filters 容器
            const exportFilters = document.querySelector('.export__type-filters');
            if (!exportFilters) {
                throw new Error('未找到导出筛选器容器');
            }
            
            // 在导出筛选器容器中查找领英区域
            const linkedinSection = exportFilters.querySelector('.t-filters__section--linkedin');
            if (!linkedinSection) {
                throw new Error('未找到导出对话框中的领英资料区域');
            }
            
            // 获取所有领英资料项
            const linkedinItems = linkedinSection.querySelectorAll('.t-filters__item');
            log(`找到 ${linkedinItems.length} 个领英资料选项`);
            
            if (linkedinItems.length === 0) {
                throw new Error('未找到领英资料选项');
            }
            
            // 点击所有未激活的领英资料项
            let selectedCount = 0;
            for (const item of linkedinItems) {
                if (!item.classList.contains('t-filters__item--active')) {
                    if (safeClick(item)) {
                        selectedCount++;
                        await wait(300); // 每个选项间短延迟
                    }
                } else {
                    selectedCount++;
                }
            }
            
            log(`已选择 ${selectedCount}/${linkedinItems.length} 个领英资料`, 'success');
            return selectedCount > 0;
            
        } catch (e) {
            log(`选择领英资料失败: ${e.message}`, 'error');
            return false;
        }
    }
    
    // 步骤5: 点击导出确认按钮
    async function clickExportConfirm() {
        log('步骤5: 点击导出确认');
        
        const selectors = [
            'button[data-test="export-prospects-confirm"]',
            'button[plerdy-tracking-id="51512084301"]'
        ];
        
        for (const selector of selectors) {
            try {
                const element = document.querySelector(selector);
                if (element) {
                    log(`找到导出确认按钮: ${selector}`);
                    if (safeClick(element)) {
                        log('导出已开始！', 'success');
                        return true;
                    }
                }
            } catch (e) {
                log(`选择器 ${selector} 失败`, 'error');
            }
        }
        
        // 查找包含"导出"的按钮（但不是"导出潜在客户"）
        const buttons = document.querySelectorAll('button');
        for (const button of buttons) {
            if (button.textContent && button.textContent.includes('导出') && 
                !button.textContent.includes('导出潜在客户')) {
                if (safeClick(button)) {
                    log('导出已开始！', 'success');
                    return true;
                }
            }
        }
        
        throw new Error('未找到导出确认按钮');
    }
    
    // 完整自动化流程
    async function runExportAutomation() {
        if (window.isExportRunning) {
            log('导出已在运行中', 'error');
            return;
        }
        
        window.isExportRunning = true;
        updateStatus('运行中', true);
        log('🚀 开始自动导出流程...');
        
        try {
            // 步骤1: 点击导出菜单
            await clickExportMenu();
            
            // 步骤2: 点击导出潜在客户（打开导出对话框）
            await clickExportLeads();
            
            // 等待导出对话框完全加载
            await wait(1000);
            
            // 步骤3: 在导出对话框中选择所有邮箱状态
            const emailResult = await selectEmailStatusesInExportDialog();
            if (!emailResult) {
                log('邮箱状态选择可能失败，继续执行...', 'error');
            }
            
            // 短暂等待
            await wait(500);
            
            // 步骤4: 在导出对话框中选择所有领英资料
            const linkedinResult = await selectLinkedInProfilesInExportDialog();
            if (!linkedinResult) {
                log('领英资料选择可能失败，继续执行...', 'error');
            }
            
            // 短暂等待
            await wait(500);
            
            // 步骤5: 点击导出确认按钮
            await clickExportConfirm();
            
            log('🎉 自动导出完成！', 'success');
            
        } catch (error) {
            log(`❌ 自动导出失败: ${error.message}`, 'error');
        } finally {
            window.isExportRunning = false;
            updateStatus('就绪', false);
        }
    }
    
    // 停止导出
    function stopExportAutomation() {
        window.isExportRunning = false;
        updateStatus('已停止', false);
        log('导出已停止', 'error');
    }
    
    // 测试函数
    async function testExportStep(step) {
        log(`测试步骤: ${step}`);
        
        switch(step) {
            case 'menu':
                return await clickExportMenu();
            case 'export':
                return await clickExportLeads();
            case 'email':
                return await selectEmailStatusesInExportDialog();
            case 'linkedin':
                return await selectLinkedInProfilesInExportDialog();
            case 'confirm':
                return await clickExportConfirm();
            default:
                log('未知测试步骤', 'error');
                return false;
        }
    }
    
    // 绑定按钮事件
    document.getElementById('run-export').addEventListener('click', runExportAutomation);
    document.getElementById('stop-export').addEventListener('click', stopExportAutomation);
    
    // 将函数暴露到全局，方便调试
    window.exportAuto = {
        run: runExportAutomation,
        stop: stopExportAutomation,
        test: testExportStep,
        log: log
    };
    
    log('✅ 自动导出面板已加载完成！', 'success');
    console.log('使用命令: exportAuto.run() 开始自动导出');
    console.log('使用命令: exportAuto.stop() 停止导出');
    console.log('测试命令: exportAuto.test("menu") - 测试点击导出菜单');
    console.log('测试命令: exportAuto.test("export") - 测试点击导出潜在客户');
    console.log('测试命令: exportAuto.test("email") - 测试选择邮箱状态');
    console.log('测试命令: exportAuto.test("linkedin") - 测试选择领英资料');
    console.log('测试命令: exportAuto.test("confirm") - 测试点击导出确认');
    
})();
